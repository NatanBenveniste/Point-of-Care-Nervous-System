#include <Arduino.h>
#include <SPI.h>
#include <Adafruit_LPS35HW.h>

// ============================================================
// PIN SETUP
// ============================================================

const int UPSTREAM_CS_PIN = 9;
const int DOWNSTREAM_CS_PIN = 17;

Adafruit_LPS35HW upstreamSensor = Adafruit_LPS35HW();
Adafruit_LPS35HW downstreamSensor = Adafruit_LPS35HW();

// ============================================================
// SETTINGS
// ============================================================

const int ZERO_SAMPLES = 150;       //750 is right
const int ZERO_DELAY_MS = 13;       // 200 * 10 ms = 2 seconds zeroing
const int SAMPLE_DELAY_MS = 13;     // about 50 Hz

// ============================================================
// VARIABLES
// ============================================================

bool collecting = false;

float upstreamZero_Pa = 0.0;
float downstreamZero_Pa = 0.0;
float deltaPZero_Pa = 0.0;

unsigned long startTime_ms = 0;

// ============================================================
// SENSOR READ FUNCTIONS
// ============================================================

float readUpstreamPa() {
  return upstreamSensor.readPressure() * 100.0;  // hPa to Pa
}

float readDownstreamPa() {
  return downstreamSensor.readPressure() * 100.0;  // hPa to Pa
}

// ============================================================
// ZERO BOTH SENSORS
// ============================================================

void calibrateZeros() {

  float sumUp = 0.0;
  float sumDown = 0.0;

  for (int i = 0; i < ZERO_SAMPLES; i++) {
    float upstream_Pa = readUpstreamPa();
    float downstream_Pa = readDownstreamPa();

    sumUp += upstream_Pa;
    sumDown += downstream_Pa;

    delay(ZERO_DELAY_MS);
  }

  upstreamZero_Pa = sumUp / ZERO_SAMPLES;
  downstreamZero_Pa = sumDown / ZERO_SAMPLES;
  deltaPZero_Pa = upstreamZero_Pa - downstreamZero_Pa;
}

// ============================================================
// SETUP
// ============================================================

void setup() {
  Serial.begin(115200);
  delay(2000);

  SPI1.setSCK(10);
  SPI1.setTX(11);
  SPI1.setRX(12);
  SPI1.begin();
  SPI.begin();

  if (!upstreamSensor.begin_SPI(UPSTREAM_CS_PIN, &SPI1)) {
    Serial.println("ERROR: Upstream LPS sensor not found.");
    while (1) {
      delay(10);
    }
  }

  if (!downstreamSensor.begin_SPI(DOWNSTREAM_CS_PIN, &SPI)) {
    Serial.println("ERROR: Downstream LPS sensor not found.");
    while (1) {
      delay(10);
    }
  }

  // Higher data rate
  upstreamSensor.setDataRate(LPS35HW_RATE_75_HZ);
  downstreamSensor.setDataRate(LPS35HW_RATE_75_HZ);

  // Internal Low-Pass Filter for cleaner readings, but adds delay
  upstreamSensor.enableLowPass(false);
  downstreamSensor.enableLowPass(false);

  Serial.println("READY");

  Serial.println("Press Enter to start.");
  Serial.println("The code will zero for 2 seconds, then begin logging.");
  Serial.println("Press Enter again to stop.");
}

// ============================================================
// LOOP
// ============================================================

void loop() {
  // ------------------------------------------------------------
  // Start / stop collection with Enter
  // ------------------------------------------------------------
  if (Serial.available() > 0) {
    Serial.readStringUntil('\n');

    collecting = !collecting;

    if (collecting) {
      Serial.println("ZEROING...");
      calibrateZeros();

      startTime_ms = millis();

      Serial.println("START_DATA");
      Serial.println("time_ms,upstream_rel_Pa,downstream_rel_Pa,deltaP_zeroed_Pa,read_gap_us");
    } else {
      Serial.println("END_DATA");
      Serial.println("Press Enter to start another test.");
    }
  }

  if (!collecting) {
    return;
  }

  // ------------------------------------------------------------
  // Read sensors and measure read timing gap
  // ------------------------------------------------------------
  unsigned long tUp_us = micros();
  float upstream_abs_Pa = readUpstreamPa();

  unsigned long tDown_us = micros();
  float downstream_abs_Pa = readDownstreamPa();

  unsigned long read_gap_us = tDown_us - tUp_us;

  // ------------------------------------------------------------
  // Calculate relative and differential pressures
  // ------------------------------------------------------------
  float upstream_rel_Pa = upstream_abs_Pa - upstreamZero_Pa;
  float downstream_rel_Pa = downstream_abs_Pa - downstreamZero_Pa;

  float deltaP_raw_Pa =  upstream_abs_Pa - downstream_abs_Pa;
  float deltaP_zeroed_Pa = deltaP_raw_Pa - deltaPZero_Pa;

  unsigned long time_ms = millis() - startTime_ms;

  // ------------------------------------------------------------
  // Print clean CSV
  // ------------------------------------------------------------
  Serial.print(time_ms);
  Serial.print(",");
  Serial.print(upstream_rel_Pa, 2);
  Serial.print(",");
  Serial.print(downstream_rel_Pa, 2);
  Serial.print(",");
  Serial.print(deltaP_zeroed_Pa, 2);
  Serial.print(",");
  Serial.println(read_gap_us);

  delay(SAMPLE_DELAY_MS);
}
